n,m=map(int,input().split())
d=[list(map(int,input().split())) for i in range(n)]

for i in range(n):
    k=0
    l=i
    for j in range(i+1):
        print(d[l][k],end=" ")
        k+=1
        l-=1
    print()
 

for  i in range(2,n+m-2):
    k=n-1
    l=i-1
    for j in range(n):
        if k>=0 and l<m:
            print(d[k][l],end=" ")
            l+=1
            k-=1
    print()
